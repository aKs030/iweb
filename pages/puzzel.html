// Puzzle-Browser-Game ähnlich wie Tetris const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const grid = 32; const rows = 20; const cols = 10; canvas.width = cols * grid; canvas.height = rows * grid; const colors = ['#FF5733', '#33FF57', '#3357FF', '#FF33A8', '#FFBD33']; // Spielfeld erstellen let board = Array.from({ length: rows }, () => Array(cols).fill(0)); // Tetromino-Formen const tetrominos = [ [[1, 1, 1, 1]], // I [[1, 1, 1], [0, 1, 0]], // T [[1, 1], [1, 1]], // O [[1, 1, 0], [0, 1, 1]], // S [[0, 1, 1], [1, 1, 0]] // Z ]; let currentPiece = { shape: tetrominos[Math.floor(Math.random() * tetrominos.length)], color: colors[Math.floor(Math.random() * colors.length)], x: 3, y: 0 }; function drawBoard() { ctx.clearRect(0, 0, canvas.width, canvas.height); for (let row = 0; row < rows; row++) { for (let col = 0; col < cols; col++) { if (board[row][col]) { ctx.fillStyle = board[row][col]; ctx.fillRect(col * grid, row * grid, grid, grid); ctx.strokeRect(col * grid, row * grid, grid, grid); } } } } function drawPiece() { ctx.fillStyle = currentPiece.color; currentPiece.shape.forEach((row, y) => { row.forEach((cell, x) => { if (cell) { ctx.fillRect((currentPiece.x + x) * grid, (currentPiece.y + y) * grid, grid, grid); ctx.strokeRect((currentPiece.x + x) * grid, (currentPiece.y + y) * grid, grid, grid); } }); }); } function gameLoop() { drawBoard(); drawPiece(); requestAnimationFrame(gameLoop); } gameLoop();
